// ─────────────────────────────────────────

const STOCKS = {
  'RELIANCE.NS': 'Reliance Industries',
  'TCS.NS':      'Tata Consultancy',
  'INFY.NS':     'Infosys',
  'HDFCBANK.NS': 'HDFC Bank',
  'WIPRO.NS':    'Wipro'
};

let activeTicker = 'RELIANCE.NS';
let mainChart = null;
let priceData = {};

// ─────────────────────────────────────────
// CLOCK
// ─────────────────────────────────────────
function updateClock() {
  const now = new Date();
  document.getElementById('clock').textContent = now.toLocaleTimeString('en-IN');
}
setInterval(updateClock, 1000);
updateClock();

// ─────────────────────────────────────────
// BUILD SIDEBAR
// ─────────────────────────────────────────
function buildSidebar() {
  const list = document.getElementById('stockList');
  list.innerHTML = '';
  Object.entries(STOCKS).forEach(([ticker, name]) => {
    const d = document.createElement('div');
    d.className = 'stock-item' + (ticker === activeTicker ? ' active' : '');
    d.id = 'sb-' + ticker;
    d.innerHTML = `
      <div class="s-name">${ticker.replace('.NS','')}</div>
      <div class="s-full">${name}</div>
      <div class="s-price" id="sb-price-${ticker}">Loading...</div>`;
    d.onclick = () => selectStock(ticker);
    list.appendChild(d);
  });
}

// ─────────────────────────────────────────
// FETCH PRICES
// ─────────────────────────────────────────
async function fetchPrices() {
  try {
    const res  = await fetch('/api/prices');
    priceData  = await res.json();

    Object.entries(priceData).forEach(([ticker, d]) => {
      if (!d.price) return;
      const pct   = d.pct >= 0 ? `+${d.pct}%` : `${d.pct}%`;
      const cls   = d.pct >= 0 ? 'up' : 'dn';
      const txt   = `₹${d.price.toLocaleString()} ${pct}`;

      // Ticker strip
      const el  = document.getElementById('t-' + ticker);
      const el2 = document.getElementById('t2-' + ticker);
      if (el)  { el.textContent  = txt; el.className  = cls; }
      if (el2) { el2.textContent = txt; el2.className = cls; }

      // Sidebar
      const sbp = document.getElementById('sb-price-' + ticker);
      if (sbp) {
        sbp.textContent = `₹${d.price.toLocaleString()} (${pct})`;
        sbp.style.color = d.pct >= 0 ? 'var(--green)' : 'var(--red)';
      }
    });

    // Update metrics for active stock
    updateMetrics();
  } catch(e) {
    console.log('Price fetch error:', e);
  }
}

function updateMetrics() {
  const d = priceData[activeTicker];
  if (!d || !d.price) return;
  const pct = d.pct >= 0 ? `+${d.pct}%` : `${d.pct}%`;
  const chg = d.change >= 0 ? `+₹${d.change}` : `-₹${Math.abs(d.change)}`;

  document.getElementById('m-price').textContent = '₹' + d.price.toLocaleString();
  document.getElementById('m-price').style.color = d.pct >= 0 ? 'var(--green)' : 'var(--red)';
  document.getElementById('m-change').textContent = `${chg} (${pct})`;
  document.getElementById('m-change').style.color = d.pct >= 0 ? 'var(--green)' : 'var(--red)';
  document.getElementById('m-high').textContent  = d.high ? '₹' + d.high.toLocaleString() : '--';
  document.getElementById('m-low').textContent   = d.low  ? '₹' + d.low.toLocaleString()  : '--';
  document.getElementById('m-vol').textContent   = d.volume ? (d.volume/100000).toFixed(1) + 'L' : '--';
  document.getElementById('m-updated').textContent = 'Updated ' + (d.updated || '--');
}

// ─────────────────────────────────────────
// CHART
// ─────────────────────────────────────────
async function loadChart(range, btn) {
  if (btn) {
    document.querySelectorAll('.range-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
  }
  try {
    const res  = await fetch(`/api/chart/${activeTicker}?range=${range}`);
    const data = await res.json();
    buildChart(data);
  } catch(e) { console.log('Chart error:', e); }
}

function buildChart(data) {
  const ctx = document.getElementById('mainChart').getContext('2d');
  if (mainChart) mainChart.destroy();

  const prices = data.prices;
  const last   = prices[prices.length - 1];
  const predicted = [...Array(prices.length - 1).fill(null), last,
                      Math.round(last * 1.018)];

  mainChart = new Chart(ctx, {
    type: 'line',
    data: {
      labels: [...data.labels, 'Next'],
      datasets: [
        {
          label: 'Actual Price (₹)',
          data: [...prices, null],
          borderColor: '#00d4ff',
          backgroundColor: 'rgba(0,212,255,0.06)',
          borderWidth: 2, pointRadius: 0, fill: true, tension: 0.3
        },
        {
          label: 'MA20',
          data: [...data.ma20, null],
          borderColor: '#ffd740',
          borderWidth: 1.5, pointRadius: 0, fill: false,
          borderDash: [4,3], tension: 0.3
        },
        {
          label: 'MA50',
          data: [...data.ma50, null],
          borderColor: '#7c3aed',
          borderWidth: 1.5, pointRadius: 0, fill: false,
          borderDash: [4,3], tension: 0.3
        },
        {
          label: 'Predicted',
          data: predicted,
          borderColor: '#00e676',
          borderDash: [6,4],
          borderWidth: 2, pointRadius: [0,0,0,0,0,6], fill: false,
          pointBackgroundColor: '#00e676', tension: 0.3
        }
      ]
    },
    options: {
      responsive: true,
      interaction: { mode: 'index', intersect: false },
      plugins: {
        legend: {
          labels: { color: '#8892a4', font: { size: 11 }, boxWidth: 20 }
        },
        tooltip: {
          backgroundColor: '#161d2e',
          borderColor: '#1e2d45', borderWidth: 1,
          titleColor: '#00d4ff', bodyColor: '#e8eaf6',
          callbacks: { label: c => ` ₹${c.parsed.y?.toLocaleString() ?? '--'}` }
        }
      },
      scales: {
        x: { grid: { color: '#1e2d45' }, ticks: { color: '#8892a4', maxTicksLimit: 8, font:{size:11} } },
        y: { grid: { color: '#1e2d45' }, ticks: { color: '#8892a4', callback: v => '₹' + v.toLocaleString(), font:{size:11} } }
      }
    }
  });
}

// ─────────────────────────────────────────
// SIGNAL
// ─────────────────────────────────────────
async function fetchSignal() {
  try {
    const res  = await fetch(`/api/signal/${activeTicker}`);
    const data = await res.json();
    const badge = document.getElementById('signalBadge');
    badge.textContent   = data.signal || 'N/A';
    badge.style.background = (data.color || '#888') + '22';
    badge.style.color      = data.color || '#888';
    badge.style.border     = `1px solid ${data.color || '#888'}`;

    const reasons = document.getElementById('signalReasons');
    reasons.innerHTML = (data.reasons || [])
      .map(r => `<span class="reason-tag">${r}</span>`).join('');
  } catch(e) {}
}

// ─────────────────────────────────────────
// INDICATORS
// ─────────────────────────────────────────
async function fetchIndicators() {
  try {
    const res  = await fetch(`/api/indicators/${activeTicker}`);
    const d    = await res.json();
    if (!d.rsi) return;

    const grid = document.getElementById('indicatorGrid');
    grid.innerHTML = `
      ${indCard('RSI', d.rsi.toFixed(1), d.rsi, 0, 100,
        d.rsi < 35 ? 'var(--green)' : d.rsi > 65 ? 'var(--red)' : 'var(--yellow)')}
      ${indCard('MACD', d.macd.toFixed(2), 50, 0, 100,
        d.macd > d.signal_ln ? 'var(--green)' : 'var(--red)')}
      ${indCard('Stoch %K', d.stoch_k.toFixed(1), d.stoch_k, 0, 100,
        d.stoch_k < 20 ? 'var(--green)' : d.stoch_k > 80 ? 'var(--red)' : 'var(--yellow)')}
      ${indCard('Vol Ratio', d.vol_ratio.toFixed(2), Math.min(d.vol_ratio * 50, 100), 0, 100,
        d.vol_ratio > 1.2 ? 'var(--green)' : 'var(--text-secondary)')}
      ${indCard('MA 20', '₹' + d.ma20.toLocaleString(), 50, 0, 100, 'var(--yellow)')}
      ${indCard('MA 50', '₹' + d.ma50.toLocaleString(), 50, 0, 100, 'var(--accent2)')}
    `;
  } catch(e) {}
}

function indCard(label, value, barPct, min, max, color) {
  const pct = Math.min(100, Math.max(0, ((barPct - min) / (max - min)) * 100));
  return `
    <div class="ind-item">
      <div class="ind-label">${label}</div>
      <div class="ind-value" style="color:${color}">${value}</div>
      <div class="ind-bar"><div class="ind-bar-fill" style="width:${pct}%;background:${color}"></div></div>
    </div>`;
}

// ─────────────────────────────────────────

async function runPredict() {
  const model  = document.getElementById('modelSel').value;
  const openP  = parseFloat(document.getElementById('openP').value)  || 0;
  const highP  = parseFloat(document.getElementById('highP').value)  || 0;
  const lowP   = parseFloat(document.getElementById('lowP').value)   || 0;

  document.getElementById('predPrice').textContent = 'Predicting...';
  document.getElementById('predResult').style.display = 'block';

  try {
    const res  = await fetch('/api/predict', {
      method: 'POST',
      headers: {'Content-Type':'application/json'},
      body: JSON.stringify({ ticker: activeTicker, model, open: openP, high: highP, low: lowP })
    });
    const data = await res.json();

    document.getElementById('predPrice').textContent = '₹' + data.predicted_price.toLocaleString();
    const chg = data.change >= 0 ? `▲ +₹${data.change} (+${data.change_pct}%)` : `▼ ₹${data.change} (${data.change_pct}%)`;
    document.getElementById('predChange').textContent = chg;
    document.getElementById('predChange').style.color = data.change >= 0 ? 'var(--green)' : 'var(--red)';
    document.getElementById('predMeta').textContent =
      `Model: ${data.model} | Accuracy: ${data.accuracy}% | Signal: ${data.signal}`;
  } catch(e) {
    document.getElementById('predPrice').textContent = 'Error';
  }
}

function selectStock(ticker) {
  activeTicker = ticker;
  document.querySelectorAll('.stock-item').forEach(el => el.classList.remove('active'));
  const el = document.getElementById('sb-' + ticker);
  if (el) el.classList.add('active');

  document.getElementById('chartTitle').textContent =
    ticker.replace('.NS','') + ' — ' + STOCKS[ticker];

  // Reset predict result
  document.getElementById('predResult').style.display = 'none';
  document.getElementById('openP').value = '';
  document.getElementById('highP').value = '';
  document.getElementById('lowP').value  = '';

  updateMetrics();
  loadChart('1w');
  fetchSignal();
  fetchIndicators();
}
